Modify build_parser() to add subparsers:

```python
def build_parser() -> argparse.ArgumentParser:
    parser = argparse.ArgumentParser(
        description="Arbiter whale copy-trading alert service",
        prog="arbiter",
    )
    parser.add_argument("--check", action="store_true", help="Validate config and connectivity, then exit")
    parser.add_argument("--verbose", "-v", action="store_true", help="Enable DEBUG logging")

    subparsers = parser.add_subparsers(dest="command")

    whales_parser = subparsers.add_parser("whales", help="Display whale rankings")
    whales_parser.add_argument("address", nargs="?", default=None, help="Show stats for a single wallet")
    whales_parser.add_argument("--all", action="store_true", dest="show_all", help="Include below-threshold wallets")
    whales_parser.add_argument(
        "--mode",
        choices=["consistent", "highroller", "frequent"],
        default=None,
        help="Scoring mode override for display only (does not update DB)",
    )
    whales_parser.add_argument("--days", type=int, default=None, help="Rolling window in days")

    return parser
```

Add display_whales function (async):

```python
async def display_whales(args: argparse.Namespace, settings) -> None:
    from sqlalchemy import select
    from arbiter.db.models import Market, Trade, Wallet
    from arbiter.scoring.whales import _apply_scores, SCORE_WEIGHTS

    engine = make_engine(settings.database_url)
    session_factory = make_session_factory(engine)
    try:
        async with session_factory() as session:
            if args.address:
                await _show_wallet_detail(session, args.address)
            else:
                await _show_whale_table(session, show_all=args.show_all, mode=args.mode, days=args.days, settings=settings)
    finally:
        await engine.dispose()
```

Add _show_whale_table:
- Query Wallet table; if not show_all, filter where is_tracked=True; order by score desc; limit 20
- If --mode differs from settings.whale_score_mode or --days is set: recompute in-memory rank scores using _apply_scores (NEVER write back to DB)
- Print header: "Rank  Address               Win Rate   Total P&L   Trades   Score"
- Print each row: rank (1-indexed), address abbreviated (first 8 chars + "..." + last 6 chars), win_rate as "X.X%" or "N/A", total_pnl formatted as "$X.XX", total_trades, score formatted to 4 decimal places
- If no rows: print "No whales found. Run the service to ingest trades and score wallets."
